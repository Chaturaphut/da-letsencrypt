#!/usr/local/bin/php short('Successfully applied SSL certificate for ' . $_POST['domain']); /** * Check if TOS is accepted and e-mail is defined */ if ($_POST['accept_tos'] !== 'on') { $log->error('You didn\'t accept the Terms of Service.'); } if (empty($_POST['email']) || !filter_var($_POST['email'], FILTER_VALIDATE_EMAIL)) { $log->error('The provided e-mailadres is invalid.'); } /** * Load account and domain */ /** @var Domain $domain */ try { $account = new Account($_SERVER['USERNAME'], $_POST['email'], 'https://acme-staging.api.letsencrypt.org/directory'); if (!$account->loadKeys()) { $account->createKeys(); $account->register(); } $domain = new Domain($_SERVER['SESSION_SELECTED_DOMAIN'], (array) $_POST['subdomains'], $account); } catch (\Exception $e) { $log->error($e->getMessage()); } /** * Request and complete challenges for domain */ $log->setErrorShort('Error while completing domain challenges'); try { $challenges = new Challenges($domain); $challenges->solveChallenge(); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully completed challenge for ' . $domain->getDomain() . '.'); /** * Request certificate */ $log->setErrorShort('Error while requesting SSL certificates'); try { $domain->createKeys(); $domain->requestCertificate(); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully received certificate from Let\'s Encrypt.'); /** * Save certificate in DirectAdmin */ $log->setErrorShort('Error while saving SSL certificates to DirectAdmin'); try { $domain->applyCertificates(); $domain->config('domain', $domain->getDomain()); $domain->config('subdomains', $domain->getSubdomains()); $domain->config('status', 'applied to DirectAdmin'); $domain->config('expire', date('Y-m-d', strtotime('+50 days'))); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully saved new certificate and CA certificates in DirectAdmin.'); $log->output();
