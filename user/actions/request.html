#!/usr/local/bin/php short('Successfully applied SSL certificate for '.$domain); // Load account and domain try { $account = new Account($_SERVER['USERNAME'], $email, 'https://acme-staging.api.letsencrypt.org/directory'); /** @var Domain $domain */ $domain = new Domain($domain, $account); } catch (\Exception $e) { $log->error($e->getMessage()); } /** * Request and complete challenges for domain */ $log->setErrorShort('Error while completing domain challenges'); try { $challenges = new Challenges($domain); $challenges->solveChallenge(); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully completed challenge for ' . $domain->getHostname() . '.'); /** * Request certificate */ $log->setErrorShort('Error while requesting SSL certificates'); try { $domain->createKeys(); $domain->requestCertificate(); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully received certificate from Let\'s Encrypt.'); /** * Save certificate in DirectAdmin */ $log->setErrorShort('Error while saving SSL certificates to DirectAdmin'); try { $domain->applyCertificates(); } catch(\Exception $e) { $log->error($e->getMessage()); } $log->info('Successfully saved new certificate and CA certificates in DirectAdmin.'); $log->output();
