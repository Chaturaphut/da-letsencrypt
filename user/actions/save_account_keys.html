#!/usr/local/bin/php short('Successfully saved account keys.'); /** * Check if both keys are defined */ if (empty($_POST['private']) || empty($_POST['public'])) { $log->error('One of the keys are not defined.'); } /** * Save both */ $account = new Account($_SERVER['USERNAME']); $account->writeToStorage('private.key', $_POST['private']); $account->writeToStorage('public.key', $_POST['public']); $log->info('Writed keys to disk.'); $log->output();
